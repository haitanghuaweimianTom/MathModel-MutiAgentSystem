# 数学建模论文全自动生成系统 — 完整重构方案

## 一、现状问题总结

当前系统存在四大致命缺陷，导致无法一次性生成可交付的完整论文：

1. **执行链断裂**：代码生成后未正确提取可执行代码，`exec()` 执行失败，结果统计全为0，论文中的数值结果全靠LLM幻觉编造
2. **上下文截断**：各阶段间传递的是截断后的文本摘要（`[:1000]`、`[:2000]`），导致模型/算法/代码/结果四者严重脱节
3. **图表缺失**：仅有文字描述的图表方案，无实际PNG图片生成，论文中所有"如图X所示"均为虚假引用
4. **字数与格式不达标**：实际中文字数仅8000-10000字，远未达到国赛正文20-30页（约16000-30000字）要求，且输出为Markdown而非标准Word格式

---

### 2.2 关键设计原则

| 原则 | 说明 |
|------|------|
| **文件引用替代文本截断** | 各阶段不传递文本摘要，而是传递文件路径，LLM通过文件读取获取完整上下文 |
| **执行验证闭环** | 代码必须实际运行并产生结构化结果，论文中的每个数据必须有来源 |
| **模型-算法-代码一致性** | 建立校验机制，确保三者描述的是同一套数学方法 |
| **图表即文件** | 图表必须是实际生成的PNG文件，而非文字描述 |
| **字数可验证** | 论文生成过程中实时统计中文字数，不足时自动触发扩展 |

**代码提取与保存**：
```python
def extract_and_save_code(raw_output: str, save_path: Path) -> str:
    """
    从LLM输出中提取纯净Python代码
    支持处理：markdown代码块、中文前缀说明、英文前缀说明
    """
    text = raw_output.strip()
    # 去除markdown代码块
    if text.startswith('```'):
        lines = text.split('\n')
        # 找到第一个和最后一个 ```
        start = 0
        while start < len(lines) and not lines[start].strip().startswith('```'):
            start += 1
        start += 1  # 跳过 ```python 行
        end = len(lines) - 1
        while end > start and not lines[end].strip().startswith('```'):
            end -= 1
        text = '\n'.join(lines[start:end]).strip()
    
    # 如果开头不是import，尝试找到第一个import的位置
    lines = text.split('\n')
    first_import = 0
    for i, line in enumerate(lines):
        if line.strip().startswith(('import ', 'from ')):
            first_import = i
            break
    
    code = '\n'.join(lines[first_import:]).strip()
    
    # 校验：必须以import开头
    if not code.startswith(('import ', 'from ')):
        raise ValueError("提取的代码不以import开头，可能包含非代码内容")
    
    save_path.write_text(code, encoding='utf-8')
    return code
```

### 3.6 Phase 5: 代码执行与验证（关键改进：隔离执行 + 自动修复）

**实现方案**：
```python
class CodeExecutor:
    def execute(self, code_path: Path, data_files: Dict, max_retries: int = 3) -> Dict:
        """
        在隔离的subprocess中执行代码
        """
        env = os.environ.copy()
        env['PYTHONPATH'] = str(code_path.parent)
        
        for attempt in range(max_retries):
            proc = subprocess.run(
                [sys.executable, str(code_path)],
                capture_output=True,
                text=True,
                timeout=300,
                env=env,
                cwd=str(code_path.parent)
            )
            
            stdout = proc.stdout
            stderr = proc.stderr
            
            # 检查 results.json 是否生成
            results_json = code_path.parent.parent / "05_execution" / "results.json"
            if results_json.exists():
                with open(results_json, 'r') as f:
                    results = json.load(f)
                
                # 验证结果有效性
                if self._validate_results(results):
                    return {
                        "success": True,
                        "results": results,
                        "stdout": stdout,
                        "stderr": stderr
                    }
            
            # 执行失败或结果无效，触发修复
            if attempt < max_retries - 1:
                fixed_code = self._fix_code(code_path, stdout, stderr)
                code_path.write_text(fixed_code, encoding='utf-8')
        
        return {"success": False, "error": stderr, "stdout": stdout}
    
    def _validate_results(self, results: Dict) -> bool:
        """验证结果是否包含有效数值"""
        if not results or not isinstance(results, dict):
            return False
        for key, value in results.items():
            if isinstance(value, dict) and 'result' in value:
                if value['result'] is not None:
                    return True
        return False
    
    def _fix_code(self, code_path: Path, stdout: str, stderr: str) -> str:
        """调用LLM修复代码"""
        original_code = code_path.read_text(encoding='utf-8')
        fix_prompt = f"""以下Python代码执行时出错，请修复后输出完整代码。

错误信息：
{stderr[:2000]}

标准输出：
{stdout[:1000]}

原始代码：
{original_code}

要求：直接输出修复后的纯Python代码，不要有任何说明文字。"""
        
        fixed = _call_llm(fix_prompt, system_prompt="你是Python调试专家。", timeout=300)
        return extract_and_save_code(fixed, code_path)
```

#### 7c. 字数保障机制

```python
def ensure_word_count(paper_md: Path, min_chars: int = 25000) -> str:
    """
    确保论文字数达标
    """
    paper_text = paper_md.read_text(encoding='utf-8')
    
    # 统计中文字符数（不含标点、空格、英文、公式）
    chinese_chars = len(re.findall(r'[一-鿿]', paper_text))
    
    if chinese_chars >= min_chars:
        return paper_text
    
    # 字数不足，触发扩展
    deficit = min_chars - chinese_chars
    
    # 找出最短的章节进行扩展
    chapters = split_into_chapters(paper_text)
    
    for chapter in sorted(chapters, key=len):
        if deficit <= 0:
            break
        
        expand_prompt = f"""请将以下论文章节进行扩充，增加详细的推导过程、分析讨论和实例说明。

原章节：
{chapter}

要求：
1. 扩充后字数增加至少1000中文字
2. 保持原有逻辑结构
3. 新增内容必须基于已有模型和结果，不得编造数据
"""
        expanded = _call_llm(expand_prompt, timeout=600)
        paper_text = paper_text.replace(chapter, expanded)
        deficit -= 1000
    
    return paper_text
```

**实现方案**：
```python
from docx import Document
from docx.shared import Pt, Inches, Cm, RGBColor
from docx.enum.text import WD_ALIGN_PARAGRAPH
from docx.oxml.ns import qn

class DocxConverter:
    def __init__(self, template_spec: Dict):
        self.spec = template_spec
    
    def convert(self, paper_md: str, charts_dir: Path, output_path: Path):
        doc = Document()
        
        # 设置页面
        section = doc.sections[0]
        section.page_width = Cm(21)
        section.page_height = Cm(29.7)
        section.top_margin = Cm(2.54)
        section.bottom_margin = Cm(2.54)
        section.left_margin = Cm(3.17)
        section.right_margin = Cm(3.17)
        
        # 解析markdown并转换
        lines = paper_md.split('\n')
        i = 0
        while i < len(lines):
            line = lines[i]
            
            if line.startswith('# '):
                # 论文标题
                self._add_title(doc, line[2:])
            elif line.startswith('## '):
                # 一级标题（如：摘要、问题重述）
                self._add_heading1(doc, line[3:])
            elif line.startswith('### '):
                # 二级标题
                self._add_heading2(doc, line[4:])
            elif line.startswith('#### '):
                # 三级标题
                self._add_heading3(doc, line[5:])
            elif line.startswith('$$') or line.startswith('$'):
                # LaTeX公式 - 使用OMath转换或图片插入
                self._add_formula(doc, line)
            elif line.startswith('!['):
                # 图片引用
                img_path = self._extract_image_path(line)
                self._add_image(doc, img_path)
            elif line.startswith('|'):
                # 表格
                table_lines = [line]
                while i + 1 < len(lines) and lines[i + 1].startswith('|'):
                    i += 1
                    table_lines.append(lines[i])
                self._add_table(doc, table_lines)
            else:
                # 普通段落
                self._add_paragraph(doc, line)
            
            i += 1
        
        # 添加页眉页脚
        self._add_header_footer(doc)
        
        doc.save(output_path)
    
    def _add_paragraph(self, doc, text):
        p = doc.add_paragraph()
        run = p.add_run(text)
        run.font.name = '宋体'
        run._element.rPr.rFonts.set(qn('w:eastAsia'), '宋体')
        run.font.size = Pt(12)
        p.paragraph_format.line_spacing = Pt(22)  # 1.5倍行距
        p.paragraph_format.first_line_indent = Cm(0.74)  # 首行缩进2字符
    
    def _add_heading1(self, doc, text):
        p = doc.add_paragraph()
        run = p.add_run(text)
        run.font.name = '黑体'
        run._element.rPr.rFonts.set(qn('w:eastAsia'), '黑体')
        run.font.size = Pt(14)
        run.bold = True
        p.alignment = WD_ALIGN_PARAGRAPH.LEFT
    
    def _add_image(self, doc, img_path: Path):
        if img_path.exists():
            doc.add_picture(str(img_path), width=Cm(14))
            last_paragraph = doc.paragraphs[-1]
            last_paragraph.alignment = WD_ALIGN_PARAGRAPH.CENTER
            # 添加图注
            caption = doc.add_paragraph()
            run = caption.add_run(f"图 X {img_path.stem}")
            run.font.name = '宋体'
            run._element.rPr.rFonts.set(qn('w:eastAsia'), '宋体')
            run.font.size = Pt(10.5)
            caption.alignment = WD_ALIGN_PARAGRAPH.CENTER
```

**LaTeX公式转换策略**：
- 简单公式：使用 `docx.oxml` 直接构建OMath结构（复杂）
- 推荐方案：使用 `latex2mathml` 转换为MathML，再用 `docx-math` 库插入
- 备选方案：调用 `matplotlib.mathtext` 将LaTeX渲染为PNG图片插入

```python
class IsolatedExecutor:
    """
    隔离的代码执行环境
    """
    def __init__(self, workspace: Path):
        self.workspace = workspace
        self.venv_path = workspace / ".venv"
    
    def setup(self):
        """创建隔离的Python虚拟环境"""
        subprocess.run([sys.executable, "-m", "venv", str(self.venv_path)], check=True)
        pip = self.venv_path / "bin" / "pip"
        subprocess.run([str(pip), "install", "numpy", "pandas", "scipy", "matplotlib", "openpyxl"], check=True)
    
    def run(self, script_path: Path, env_vars: Dict = None) -> subprocess.CompletedProcess:
        """在隔离环境中运行脚本"""
        python = self.venv_path / "bin" / "python"
        env = os.environ.copy()
        if env_vars:
            env.update(env_vars)
        
        return subprocess.run(
            [str(python), str(script_path)],
            capture_output=True,
            text=True,
            timeout=300,
            env=env,
            cwd=str(script_path.parent)
        )
```
