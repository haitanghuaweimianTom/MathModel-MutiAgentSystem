## 图表实现代码示例

以下是使用Python matplotlib生成上述图表的代码框架：

```python
import matplotlib.pyplot as plt
import numpy as np
from mpl_toolkits.mplot3d import Axes3D

# 设置学术图表风格
plt.style.use('seaborn-v0_8-whitegrid')
plt.rcParams['font.size'] = 10
plt.rcParams['axes.labelsize'] = 11
plt.rcParams['axes.titlesize'] = 12
plt.rcParams['figure.dpi'] = 150

# 图1：遮蔽时长与问题难度对比
fig, ax1 = plt.subplots()
problems = ['问题1', '问题2', '问题3', '问题4', '问题5']
shelter_time = [2.5, 3.1, 7.8, 9.2, 15.6]
difficulty = [2, 3, 4, 4, 5]

colors = ['#2E86AB', '#3B9ECF', '#5BB0D0', '#7DC3E0', '#A0D7F0']
bars = ax1.bar(problems, shelter_time, color=colors, edgecolor='black', linewidth=0.5)
ax1.set_ylabel('有效遮蔽时长 (s)', color='#2E86AB')
ax1.tick_params(axis='y', labelcolor='#2E86AB')

ax2 = ax1.twinx()
ax2.plot(problems, difficulty, 'o-', color='#A23B72', linewidth=2, markersize=8)
ax2.set_ylabel('难度等级', color='#A23B72')
ax2.tick_params(axis='y', labelcolor='#A23B72')

plt.title('各问题遮蔽时长与难度对比')
fig.tight_layout()
plt.savefig('fig1_shelter_difficulty.png', dpi=300, bbox_inches='tight')
```
